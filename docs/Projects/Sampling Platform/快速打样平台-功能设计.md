```python
    _name = 'sample.order.line'

    material_id = fields.Many2one('material.material', '材料', required=True)
    material_category = fields.Many2one(related='material_id.category_id', string='材料分类', store=True)
    
    # 尺寸信息
    length = fields.Float('长度(mm)', digits=(10, 2), required=True)
    width = fields.Float('宽度(mm)', digits=(10, 2), required=True)
    thickness = fields.Float('厚度(μm)', digits=(10, 2))
    
    # 数量信息
    quantity = fields.Integer('数量', default=1, required=True)
    area = fields.Float('面积(cm²)', compute='_compute_area', store=True)
    
    # 加工信息
    process_type = fields.Selection([
        ('die_cut', '模切'),
        ('lamination', '贴合'),
        ('printing', '印刷'),
        ('slitting', '分切'),
    ], string='加工类型', required=True)
    
    special_requirements = fields.Text('特殊要求')
    
    # 费用信息
    material_cost = fields.Monetary('材料成本', compute='_compute_costs', store=True)
    processing_cost = fields.Monetary('加工费用', compute='_compute_costs', store=True)
    unit_price = fields.Monetary('单价', compute='_compute_costs', store=True)
    subtotal = fields.Monetary('小计', compute='_compute_costs', store=True)
    currency_id = fields.Many2one(related='order_id.currency_id')
    
    @api.depends('length', 'width')
    def _compute_area(self):
        for record in self:
            record.area = (record.length * record.width) / 100  # 转换为cm²
    
    @api.depends('material_id', 'area', 'quantity', 'process_type')
    def _compute_costs(self):
        for record in self:
            # 材料成本 = 材料单价 * 面积 * 数量
            material_price = record.material_id.reference_price or 0
            record.material_cost = material_price * record.area * record.quantity / 10000
            
            # 加工费用(根据加工类型)
            process_rates = {
                'die_cut': 0.05,  # 每cm² 0.05元
                'lamination': 0.08,
                'printing': 0.10,
                'slitting': 0.03,
            }
            rate = process_rates.get(record.process_type, 0.05)
            record.processing_cost = record.area * record.quantity * rate
            
            # 单价和小计
            record.unit_price = (record.material_cost + record.processing_cost) / record.quantity if record.quantity else 0
            record.subtotal = record.material_cost + record.processing_cost
```

```javascript
class SampleDesigner {
    constructor(canvas) {
        this.canvas = canvas;
        this.layers = [];
        this.materials = [];
    }
    
    addLayer(material, dimensions) {
        const layer = {
            id: this.layers.length + 1,
            material: material,
            length: dimensions.length,
            width: dimensions.width,
            thickness: material.thickness,
            position: {x: 0, y: 0}
        };
        this.layers.push(layer);
        this.render();
        this.calculateCost();
    }
    
    calculateCost() {
        let totalCost = 0;
        this.layers.forEach(layer => {
            const area = (layer.length * layer.width) / 100;
            const materialCost = layer.material.price * area / 10000;
            const processCost = area * 0.05;
            totalCost += materialCost + processCost;
        });
        return totalCost;
    }
    
    exportDesign() {
        return {
            layers: this.layers,
            totalThickness: this.layers.reduce((sum, l) => sum + l.thickness, 0),
            estimatedCost: this.calculateCost()
        };
    }
}
```

### 打样费用结构

```python
def calculate_sample_fee(order):
    """
    打样费用计算
    """
    # 基础费用
    base_fee = 100
    
    # 材料成本
    material_cost = sum(line.material_cost for line in order.line_ids)
    
    # 加工费用
    processing_cost = sum(line.processing_cost for line in order.line_ids)
    
    # 紧急程度加价
    urgency_multiplier = {
        'normal': 1.0,      # 3-5天
        'urgent': 1.5,      # 1-2天
        'super_urgent': 2.0 # 24小时
    }
    multiplier = urgency_multiplier.get(order.urgency, 1.0)
    
    # 数量折扣
    quantity_discount = 1.0
    if order.quantity >= 50:
        quantity_discount = 0.9
    elif order.quantity >= 100:
        quantity_discount = 0.8
    
    # 总费用
    total = (base_fee + material_cost + processing_cost) * multiplier * quantity_discount
    
    return {
        'base_fee': base_fee,
        'material_cost': material_cost,
        'processing_cost': processing_cost,
        'sample_fee': base_fee * multiplier,
        'total': total
    }
```

| 服务类型 | 交期 | 基础费 | 加价 |
|---------|------|--------|------|

| 加急打样 | 1-2天 | ¥100 | 1.5x |
| 特急打样 | 24小时 | ¥100 | 2.0x |

**材料成本:** 实际材料价格 × 用量  
**加工费用:** 根据加工类型和面积计算

```python
def _create_production_order(self):
    """从打样订单创建生产工单"""
    # 创建 BOM(物料清单)
    bom = self.env['mrp.bom'].create({
        'product_tmpl_id': self.product_id.product_tmpl_id.id,
        'product_qty': self.quantity,
        'type': 'normal',
        'bom_line_ids': [(0, 0, {
            'product_id': line.material_id.product_id.id,
            'product_qty': line.quantity,
        }) for line in self.line_ids],
    })
    
    # 创建生产工单
    production = self.env['mrp.production'].create({
        'product_id': self.product_id.id,
        'product_qty': self.quantity,
        'bom_id': bom.id,
        'origin': self.name,
        'date_planned_start': fields.Datetime.now(),
        'date_planned_finished': self.promised_date,
    })
    
    self.production_order_id = production.id
    production.action_confirm()
    
    return production
```
