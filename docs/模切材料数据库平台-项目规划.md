# 模切材料数据库平台 - 项目规划方案

## 📋 项目概述

### 项目名称
**模切材料智能数据库平台 (Die-Cut Material Database Platform)**

### 项目定位
为模切行业提供一站式材料信息管理和查询服务的 SaaS 平台

### 目标用户
- 模切加工企业
- 材料供应商
- 产品设计师
- 采购人员

### 核心价值
- 解决材料信息分散、查找困难的痛点
- 提供智能材料推荐和搭配方案
- 降低材料选型成本和时间
- 建立行业材料数据标准

class MaterialCategory(models.Model):
    _name = 'material.category'
    _description = '材料分类'
    _parent_name = 'parent_id'
    _parent_store = True
    _order = 'complete_name'
    
    name = fields.Char('分类名称', required=True)

    description = fields.Text('描述')
    image = fields.Binary('分类图片')

#### 2. 材料模型 (material.material)

```python
class Material(models.Model):
    _name = 'material.material'
    _description = '材料'
    _inherit = ['mail.thread', 'mail.activity.mixin']
    
    # 基本信息
    name = fields.Char('材料名称', required=True, tracking=True)
    code = fields.Char('材料编号', required=True, copy=False, tracking=True)
    category_id = fields.Many2one('material.category', '材料分类', required=True)
    manufacturer_id = fields.Many2one('res.partner', '生产商', 
                                      domain="[('is_company', '=', True), ('supplier_rank', '>', 0)]")
    brand = fields.Char('品牌')
    origin = fields.Char('产地')
    state = fields.Selection([
        ('active', '在产'),
        ('discontinued', '停产'),
        ('development', '开发中'),
    ], string='状态', default='active', tracking=True)
    
    # 物理特性
    thickness = fields.Float('厚度(μm)', digits=(10, 2))
    width = fields.Float('宽度(mm)', digits=(10, 2))
    length = fields.Float('长度(m)', digits=(10, 2))
    density = fields.Float('密度(g/cm³)', digits=(10, 3))
    color = fields.Char('颜色')
    transparency = fields.Selection([
        ('transparent', '透明'),
        ('translucent', '半透明'),
        ('opaque', '不透明'),
    ], string='透明度')
    
    # 性能参数
    tensile_strength = fields.Float('拉伸强度(MPa)', digits=(10, 2))
    tear_strength = fields.Float('撕裂强度(N)', digits=(10, 2))
    temp_resistance_min = fields.Float('耐温下限(℃)')
    temp_resistance_max = fields.Float('耐温上限(℃)')
    adhesion = fields.Float('粘性(N/25mm)', digits=(10, 2))
    flame_rating = fields.Selection([
        ('v0', 'V-0'),
        ('v1', 'V-1'),
        ('v2', 'V-2'),
        ('hb', 'HB'),
    ], string='阻燃等级')
    
    # 应用信息
    application_ids = fields.Many2many('material.application', string='应用场景')
    industry_ids = fields.Many2many('material.industry', string='适用行业')
    process_note = fields.Text('加工工艺说明')
    caution = fields.Text('注意事项')
    
    # 商务信息
    supplier_ids = fields.Many2many('res.partner', string='供应商',
                                    domain="[('is_company', '=', True), ('supplier_rank', '>', 0)]")
    reference_price = fields.Monetary('参考价格', currency_field='currency_id')
    currency_id = fields.Many2one('res.currency', string='货币', 
                                   default=lambda self: self.env.company.currency_id)
    moq = fields.Float('最小起订量', digits=(10, 2))
    moq_unit = fields.Many2one('uom.uom', '起订量单位')
    lead_time = fields.Integer('交货期(天)')
    certification_ids = fields.Many2many('material.certification', string='认证')
    
    # 文档资料
    document_ids = fields.One2many('material.document', 'material_id', '文档资料')
    image_ids = fields.One2many('material.image', 'material_id', '产品图片')
    
    # 关联信息
    combination_ids = fields.Many2many('material.combination', string='推荐搭配')
    alternative_ids = fields.Many2many('material.material', 
                                       'material_alternative_rel',
                                       'material_id', 'alternative_id',
                                       string='替代材料')
    
    # 统计信息
    view_count = fields.Integer('浏览次数', default=0)
    inquiry_count = fields.Integer('询价次数', default=0)
    rating = fields.Float('评分', digits=(2, 1))
    
    # 其他
    description = fields.Html('详细描述')
    active = fields.Boolean('有效', default=True)
```

```python
class Material(models.Model):
    _name = 'material.material'
    
    @api.model
    def advanced_search(self, search_params):
        """
        高级搜索功能
        search_params = {
            'category_id': 分类ID,
            'thickness_min': 最小厚度,
            'thickness_max': 最大厚度,
            'temp_resistance_min': 最低耐温,
            'application_ids': 应用场景IDs,
            'industry_ids': 行业IDs,
            'manufacturer_id': 生产商ID,
            'price_max': 最高价格,
            ...
        }
        """
        domain = []
        
        if search_params.get('category_id'):
            domain.append(('category_id', 'child_of', search_params['category_id']))
        
        if search_params.get('thickness_min'):
            domain.append(('thickness', '>=', search_params['thickness_min']))
        
        if search_params.get('thickness_max'):
            domain.append(('thickness', '<=', search_params['thickness_max']))
        
        if search_params.get('temp_resistance_min'):
            domain.append(('temp_resistance_min', '<=', search_params['temp_resistance_min']))
            domain.append(('temp_resistance_max', '>=', search_params['temp_resistance_min']))
        
        if search_params.get('application_ids'):
            domain.append(('application_ids', 'in', search_params['application_ids']))
        
        # ... 更多筛选条件
        
        return self.search(domain)
```
